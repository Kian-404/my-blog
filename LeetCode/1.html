/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var twoSum = function(nums, target) {
  let i = nums.length;
  while(i > 1) {
      let last = nums.pop();
      if (nums.indexOf(target - last) > -1) {
          return [nums.indexOf(target - last), nums.length]
      }
      i--
  }
};
